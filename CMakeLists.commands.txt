find_program(CTEST_EXECUTABLE ctest)
if(NOT CTEST_EXECUTABLE)
    message(FATAL_ERROR "ctest not found!")
endif()



add_custom_target(build
  COMMAND ${CMAKE_MAKE_PROGRAM} -j8
)

add_custom_target(run
    COMMAND ${CMAKE_BINARY_DIR}/CloverDemo
    DEPENDS build CloverDemo
    WORKING_DIRECTORY ${CMAKE_BINARY_DIR}
    USES_TERMINAL
)

find_program(
  CLANG_FORMAT_BIN clang-format
)
if(CLANG_FORMAT_BIN)
  file(GLOB_RECURSE FORMAT_SOURCE_LIST
    src/*.cpp src/*.h
    includes/*.cpp includes/*.h
    test/*.cc
  )

  # Create custom target
  add_custom_target(
    fmt
    COMMAND ${CLANG_FORMAT_BIN}
    -i
    -style=file
    ${FORMAT_SOURCE_LIST}
    WORKING_DIRECTORY ${CMAKE_SOURCE_DIR}
  )
else()
    add_custom_target(
        fmt
        COMMAND ${CMAKE_COMMAND} -E echo "clang-format binary not found."
    )
endif()

find_program(CLOC_BIN cloc)
if(CLOC_BIN)
    add_custom_target(run_cloc
        COMMAND cloc . --fullpath --exclude-dir=dependencies,build,.vscode,docs
        WORKING_DIRECTORY ${CMAKE_SOURCE_DIR}
    )
else()
    add_custom_target(run_cloc
        COMMAND ${CMAKE_COMMAND} -E echo "cloc binary not found."
    )
endif()

find_program(PPROF_BIN pprof)
find_file(PROF_SO_FILE /usr/local/lib/prof.so)

if(PROF_SO_FILE AND PPROF_BIN)
  add_custom_target(profile_call_graph
    DEPENDS build
    COMMAND env LD_PRELOAD=${PROF_SO_FILE} CPUPROFILE_FREQUENCY=48000 CPUPROFILE=./test.prof ${CMAKE_BINARY_DIR}/CloverDemo
    COMMAND ${PPROF_BIN} -web -cum ${CMAKE_BINARY_DIR}/CloverDemo ./test.prof
  )
else()
  add_custom_target(profile_call_graph
    COMMAND ${CMAKE_COMMAND} -E echo "Either /usr/local/lib/prof.so or pprof not found."
  )
endif()
